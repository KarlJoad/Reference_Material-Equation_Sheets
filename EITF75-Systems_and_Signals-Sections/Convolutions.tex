\section{Convolutions}\label{sec:Convolutions}
\begin{definition}[Convolution]\label{def:Convolution}
  The \emph{convolution} operator.

  \begin{equation}\label{eq:Convolution}
    y(t) = \sum\limits_{k=-\infty}^{\infty} x(k) * h(n-k)
  \end{equation}
\end{definition}
