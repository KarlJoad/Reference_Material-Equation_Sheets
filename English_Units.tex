\DeclareSIUnit\yd{yd}

\DeclareSIUnit\pt{pt}

\DeclareSIUnit\qt{qt}

\DeclareSIUnit\gallon{gal}
\DeclareSIUnit\gal{gal}

% Mass
\DeclareSIUnit\grain{gr}
\DeclareSIUnit\gr{gr}

\DeclareSIUnit\ounce{oz}
\DeclareSIUnit\oz{oz}
